{{{

}}}
* Should be: 
{{{
student_tuple = ('Amanda', 'Blue', [98, 75, 87])

student_tuple = ('Amanda', 'Blue', [98, 75, 87])
lst = list(student_tuple)
lst[0] = 'Ariana'
student_tuple = tuple( lst )
student_tuple
}}}
